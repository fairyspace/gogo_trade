### 1.1 六大核心职能

```
金融CRM职能地图

├── 职能①：客户全生命周期管理
│   ├── 获客    → 渠道追踪、线索管理、开户转化
│   ├── 活客    → 行为激活、产品体验、交易促进
│   ├── 留客    → 流失预警、挽留策略、权益体系
│   └── 升客    → 资产提升、产品升级、VIP培育
│
├── 职能②：客户画像与洞察
│   ├── 基础属性   → 人口统计、风险偏好、资产规模
│   ├── 行为画像   → 交易习惯、内容偏好、活跃规律
│   ├── 价值画像   → CLV评分、贡献度、潜力预测
│   └── 风险画像   → 风险承受、持仓风险、信用风险
│
├── 职能③：智能推荐引擎
│   ├── 产品推荐   → 基金/股票/债券/理财 个性化推荐
│   ├── 内容推荐   → 研报/资讯/直播 个性化分发
│   ├── 服务推荐   → 投顾服务/增值服务 匹配推荐
│   └── 活动推荐   → 个性化活动匹配
│
├── 职能④：营销自动化
│   ├── 人群圈选   → 规则+模型双驱动
│   ├── 策略编排   → 多渠道、多触点、自动化流程
│   ├── 内容生产   → 个性化文案、动态内容
│   └── 效果归因   → 多触点归因分析
│
├── 职能⑤：客服与销售支持
│   ├── 客户360视图  → 坐席实时查看客户全貌
│   ├── 销售线索     → 线索分配、跟进管理
│   ├── 服务工单     → 投诉/咨询 全流程追踪
│   └── 智能辅助     → 话术推荐、产品推荐辅助
│
└── 职能⑥：数据分析与决策支持
    ├── 经营看板   → 核心KPI实时监控
    ├── 客户分析   → 分层分群、趋势分析
    ├── 活动分析   → 营销ROI、效果归因
    └── 预测模型   → 流失预测、价值预测
```

## 三、完整表结构设计

### 3.1 客户基础域

```sql
-- ① 客户基础信息表
CREATE TABLE `crm_customer_base` (
  `customer_id`       BIGINT       NOT NULL COMMENT '客户ID(全局唯一)',
  `customer_no`       VARCHAR(32)  NOT NULL COMMENT '客户编号(业务号)',
  `mobile`            VARCHAR(11)  NOT NULL COMMENT '手机号(脱敏存储)',
  `mobile_hash`       VARCHAR(64)  NOT NULL COMMENT '手机号Hash(用于查询)',
  `real_name`         VARCHAR(50)  COMMENT '真实姓名(加密存储)',
  `id_card_hash`      VARCHAR(64)  COMMENT '身份证Hash',
  `gender`            TINYINT      COMMENT '性别 0-未知 1-男 2-女',
  `birth_date`        DATE         COMMENT '出生日期',
  `age_range`         TINYINT      COMMENT '年龄段 1-18~25 2-26~35 3-36~45 4-46~55 5-55+',
  `city_code`         VARCHAR(10)  COMMENT '城市编码',
  `province_code`     VARCHAR(10)  COMMENT '省份编码',
  `education`         TINYINT      COMMENT '学历 1-高中 2-大专 3-本科 4-研究生 5-博士',
  `occupation`        TINYINT      COMMENT '职业类型 1-工薪 2-个体 3-企业主 4-退休',
  `customer_type`     TINYINT      COMMENT '客户类型 1-个人 2-机构',
  `customer_level`    TINYINT      COMMENT '客户等级 1-普通 2-银 3-金 4-白金 5-私行',
  `register_channel`  VARCHAR(32)  COMMENT '注册渠道 App/Web/小程序/线下',
  `register_source`   VARCHAR(64)  COMMENT '注册来源(UTM参数)',
  `referrer_id`       BIGINT       COMMENT '推荐人ID',
  `register_time`     DATETIME     COMMENT '注册时间',
  `activate_time`     DATETIME     COMMENT '激活时间',
  `open_account_time` DATETIME     COMMENT '开户时间',
  `first_trade_time`  DATETIME     COMMENT '首次交易时间',
  `status`            TINYINT      DEFAULT 1 COMMENT '状态 0-注销 1-正常 2-冻结',
  `kyc_status`        TINYINT      DEFAULT 0 COMMENT 'KYC状态 0-未完成 1-基础 2-完整',
  `create_time`       DATETIME     DEFAULT CURRENT_TIMESTAMP,
  `update_time`       DATETIME     DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`customer_id`),
  UNIQUE KEY `uk_customer_no` (`customer_no`),
  KEY `idx_mobile_hash` (`mobile_hash`),
  KEY `idx_register_time` (`register_time`),
  KEY `idx_customer_level` (`customer_level`)
) ENGINE=InnoDB COMMENT='客户基础信息表';


-- ② 客户账户信息表
CREATE TABLE `crm_customer_account` (
  `account_id`        BIGINT       NOT NULL COMMENT '账户ID',
  `customer_id`       BIGINT       NOT NULL COMMENT '客户ID',
  `account_no`        VARCHAR(32)  NOT NULL COMMENT '账户号',
  `account_type`      TINYINT      COMMENT '账户类型 1-证券 2-基金 3-期货 4-港股',
  `broker_id`         VARCHAR(16)  COMMENT '券商ID',
  `open_date`         DATE         COMMENT '开户日期',
  `open_branch`       VARCHAR(64)  COMMENT '开户营业部',
  `open_channel`      VARCHAR(32)  COMMENT '开户渠道',
  `total_asset`       DECIMAL(20,4) COMMENT '总资产(元)',
  `available_cash`    DECIMAL(20,4) COMMENT '可用资金',
  `market_value`      DECIMAL(20,4) COMMENT '持仓市值',
  `frozen_amount`     DECIMAL(20,4) COMMENT '冻结金额',
  `profit_loss`       DECIMAL(20,4) COMMENT '浮动盈亏',
  `profit_loss_rate`  DECIMAL(10,6) COMMENT '浮动盈亏率',
  `cum_profit`        DECIMAL(20,4) COMMENT '累计盈亏',
  `asset_update_time` DATETIME     COMMENT '资产更新时间',
  `status`            TINYINT      DEFAULT 1 COMMENT '账户状态 1-正常 2-冻结 3-注销',
  `create_time`       DATETIME     DEFAULT CURRENT_TIMESTAMP,
  `update_time`       DATETIME     DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`account_id`),
  KEY `idx_customer_id` (`customer_id`),
  KEY `idx_account_no` (`account_no`)
) ENGINE=InnoDB COMMENT='客户账户信息表';


-- ③ 客户风险画像表
CREATE TABLE `crm_customer_risk_profile` (
  `id`                  BIGINT    NOT NULL AUTO_INCREMENT,
  `customer_id`         BIGINT    NOT NULL COMMENT '客户ID',
  `risk_level`          TINYINT   COMMENT '风险等级 1-保守 2-稳健 3-平衡 4-积极 5-激进',
  `risk_score`          INT       COMMENT '风险评测得分',
  `assessment_version`  VARCHAR(16) COMMENT '测评版本号',
  `assessment_time`     DATETIME  COMMENT '测评时间',
  `assessment_expire`   DATETIME  COMMENT '测评有效期',
  `assessment_channel`  VARCHAR(32) COMMENT '测评渠道',
  `investment_exp`      TINYINT   COMMENT '投资经验年限 1-<1年 2-1~3年 3-3~5年 4->5年',
  `investment_target`   TINYINT   COMMENT '投资目标 1-保值 2-稳健增值 3-增值 4-高收益',
  `loss_tolerance`      TINYINT   COMMENT '可承受最大亏损 1-5% 2-10% 3-20% 4-30% 5-不限',
  `annual_income`       TINYINT   COMMENT '年收入范围',
  `total_asset_range`   TINYINT   COMMENT '总资产范围',
  `investment_period`   TINYINT   COMMENT '投资期限偏好 1-<1年 2-1~3年 3->3年',
  `is_effective`        TINYINT   DEFAULT 1 COMMENT '是否有效 1-有效 0-失效',
  `create_time`         DATETIME  DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_customer_id` (`customer_id`),
  KEY `idx_assessment_time` (`assessment_time`)
) ENGINE=InnoDB COMMENT='客户风险画像表';
```

### 3.2 标签画像域

```sql
-- ④ 标签定义表
CREATE TABLE `crm_tag_definition` (

  `tag_code`        VARCHAR(64)  NOT NULL COMMENT '标签编码(唯一)',
  `tag_name`        VARCHAR(128) NOT NULL COMMENT '标签名称',

  `tag_value`       VARCHAR(256) COMMENT '标签值',

  `confidence`      DECIMAL(5,4) COMMENT '置信度 0-1',
  `source`          VARCHAR(32)  COMMENT '来源 rule/model/manual',
  `source_detail`   VARCHAR(256) COMMENT '来源详情',
  `effective_date`  DATE         COMMENT '生效日期',
  `expire_date`     DATE         COMMENT '失效日期',

  KEY `idx_customer_id` (`customer_id`)
```

> ## 设计总结

> | 维度 | 设计决策 |
> |------|---------|
> | **核心职能** | 识别→理解→服务→增长客户 四大闭环 |

> | **合规设计** | 适当性校验嵌入推荐链路，强制不可绕过 |
